219 void StoreList::sortList()
220 {
221  qDebug()<< "TRYING TO SORT LIST" << this->list.size();
222  qSort(this->list.begin(), this->list.begin(), [this](const FMH::MODEL& e1, const FMH::MODEL& e2) -> bool
223  {
224  qDebug()<< "TRYIT LIST";
225 
226  auto role = static_cast<FMH::MODEL_KEY>(this->sortBy);;
227 
228  switch(role)
229  {
230  case FMH::MODEL_KEY::RATE:
231  case FMH::MODEL_KEY::COUNT:
232  {
233  if(e1[role].toDouble() > e2[role].toDouble())
234  return true;
235  break;
236  }
237 
238  case FMH::MODEL_KEY::MODIFIED:
239  case FMH::MODEL_KEY::DATE:
240  {
241  auto currentTime = QDateTime::currentDateTime();
242 
243  auto date1 = QDateTime::fromString(e1[role], Qt::TextDate);
244  auto date2 = QDateTime::fromString(e2[role], Qt::TextDate);
245 
246  if(date1.secsTo(currentTime) < date2.secsTo(currentTime))
247  return true;
248 
249  break;
250  }
251 
252  case FMH::MODEL_KEY::LABEL:
253  case FMH::MODEL_KEY::USER:
254  case FMH::MODEL_KEY::OWNER:
255  {
256  const auto str1 = QString(e1[role]).toLower();
257  const auto str2 = QString(e2[role]).toLower();
258 
259  if(str1 < str2)
260  return true;
261  break;
262  }
263 
264  default:
265  if(e1[role] < e2[role])
266  return true;
267  }
268 
269  return false;
270  });
271 }
