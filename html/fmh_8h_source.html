462  inline QVariantMap dirConf(const QString &path)
463  {
464  if(!FMH::fileExists(path))
465  return QVariantMap();
466 
467  QString icon, iconsize, hidden, detailview, showthumbnail, showterminal;
468  uint count = 0, sortby = FMH::MODEL_KEY::MODIFIED, viewType = 0;
469  bool foldersFirst = false;
470 
471  #ifdef Q_OS_ANDROID
472  QSettings file(path, QSettings::Format::NativeFormat);
473  file.beginGroup(QString("Desktop Entry"));
474  icon = file.value("Icon").toString();
475  file.endGroup();
476 
477  file.beginGroup(QString("Settings"));
478  hidden = file.value("HiddenFilesShown").toString();
479  file.endGroup();
480 
481  file.beginGroup(QString("MAUIFM"));
482  iconsize = file.value("IconSize").toString();
483  detailview = file.value("DetailView").toString();
484  showthumbnail = file.value("ShowThumbnail").toString();
485  showterminal = file.value("ShowTerminal").toString();
486  count = file.value("Count").toInt();
487  sortby = file.value("SortBy").toInt();
488  foldersFirst = file.value("FoldersFirst").toBool();
489  viewType = file.value("ViewType").toInt();
490  file.endGroup();
491 
492  #else
493  KConfig file(path);
494  icon = file.entryMap(QString("Desktop Entry"))["Icon"];
495  hidden = file.entryMap(QString("Settings"))["HiddenFilesShown"];
496  iconsize = file.entryMap(QString("MAUIFM"))["IconSize"];
497  detailview = file.entryMap(QString("MAUIFM"))["DetailView"];
498  showthumbnail = file.entryMap(QString("MAUIFM"))["ShowThumbnail"];
499  showterminal = file.entryMap(QString("MAUIFM"))["ShowTerminal"];
500  count = file.entryMap(QString("MAUIFM"))["Count"].toInt();
501  sortby = file.entryMap(QString("MAUIFM"))["SortBy"].toInt();
502  foldersFirst = file.entryMap(QString("MAUIFM"))["FoldersFirst"] == "true" ? true : false;
503  viewType = file.entryMap(QString("MAUIFM"))["ViewType"].toInt();
504  #endif
505 
506  auto res = QVariantMap({
507  {FMH::MODEL_NAME[FMH::MODEL_KEY::ICON], icon.isEmpty() ? "folder" : icon},
508  {FMH::MODEL_NAME[FMH::MODEL_KEY::ICONSIZE], iconsize},
509  {FMH::MODEL_NAME[FMH::MODEL_KEY::COUNT], count},
510  {FMH::MODEL_NAME[FMH::MODEL_KEY::SHOWTERMINAL], showterminal.isEmpty() ? "false" : showterminal},
511  {FMH::MODEL_NAME[FMH::MODEL_KEY::SHOWTHUMBNAIL], showthumbnail.isEmpty() ? "false" : showthumbnail},
512  {FMH::MODEL_NAME[FMH::MODEL_KEY::DETAILVIEW], detailview.isEmpty() ? "false" : detailview},
513  {FMH::MODEL_NAME[FMH::MODEL_KEY::HIDDEN], hidden.isEmpty() ? false : (hidden == "true" ? true : false)},
514  {FMH::MODEL_NAME[FMH::MODEL_KEY::SORTBY], sortby},
515  {FMH::MODEL_NAME[FMH::MODEL_KEY::FOLDERSFIRST], foldersFirst},
516  {FMH::MODEL_NAME[FMH::MODEL_KEY::VIEWTYPE], viewType}
517  });
518 
519  return res;
520  }

537  inline QString getIconName(const QString &path)
538  {
539  if(QFileInfo(path).isDir())
540  {
541  if(folderIcon.contains(path))
542  return folderIcon[path];
543  else
544  {
545  auto icon = FMH::dirConf(QString(path+"/%1").arg(".directory"))[FMH::MODEL_NAME[FMH::MODEL_KEY::ICON]].toString();
546  return icon.isEmpty() ? "folder" : icon;
547  }
548 
549  }else {
550 
551  #if defined(Q_OS_ANDROID)
552  QMimeDatabase mime;
553  auto type = mime.mimeTypeForFile(path);
554  return type.iconName();
555  #else
556  KFileItem mime(path);
557  return mime.iconName();
558  #endif
559  }
560  }
