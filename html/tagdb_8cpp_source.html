60 void TAGDB::prepareCollectionDB() const
61 {
62  QSqlQuery query(this->m_db);
63 
64  QFile file(":/script.sql");
65 
66  if (!file.exists())
67  {
68  QString log = QStringLiteral("Fatal error on build database. The file '");
69  log.append(file.fileName() + QStringLiteral("' for database and tables creation query cannot be not found!"));
70  qDebug()<<log;
71  return;
72  }
73 
74  if (!file.open(QIODevice::ReadOnly))
75  {
76  qDebug()<<QStringLiteral("Fatal error on try to create database! The file with sql queries for database creation cannot be opened!");
77  return;
78  }
79 
80  bool hasText;
81  QString line;
82  QByteArray readLine;
83  QString cleanedLine;
84  QStringList strings;
85 
86  while (!file.atEnd())
87  {
88  hasText = false;
89  line = "";
90  readLine = "";
91  cleanedLine = "";
92  strings.clear();
93  while (!hasText)
94  {
95  readLine = file.readLine();
96  cleanedLine = readLine.trimmed();
97  strings = cleanedLine.split("--");
98  cleanedLine = strings.at(0);
99  if (!cleanedLine.startsWith("--") && !cleanedLine.startsWith("DROP") && !cleanedLine.isEmpty())
100  line += cleanedLine;
101  if (cleanedLine.endsWith(";"))
102  break;
103  if (cleanedLine.startsWith("COMMIT"))
104  hasText = true;
105  }
106  if (!line.isEmpty())
107  {
108  if (!query.exec(line))
109  {
110  qDebug()<<"exec failed"<<query.lastQuery()<<query.lastError();
111  }
112 
113  } else qDebug()<<"exec wrong"<<query.lastError();
114  }
115  file.close();
116 }

149 bool TAGDB::insert(const QString &tableName, const QVariantMap &insertData)
150 {
151  if (tableName.isEmpty())
152  {
153  qDebug()<<QStringLiteral("Fatal error on insert! The table name is empty!");
154  return false;
155 
156  } else if (insertData.isEmpty())
157  {
158  qDebug()<<QStringLiteral("Fatal error on insert! The insertData is empty!");
159  return false;
160  }
161 
162  QStringList strValues;
163  QStringList fields = insertData.keys();
164  QVariantList values = insertData.values();
165  int totalFields = fields.size();
166  for (int i = 0; i < totalFields; ++i)
167  strValues.append("?");
168 
169  QString sqlQueryString = "INSERT INTO " + tableName + " (" + QString(fields.join(",")) + ") VALUES(" + QString(strValues.join(",")) + ")";
170  QSqlQuery query(this->m_db);
171  query.prepare(sqlQueryString);
172 
173  int k = 0;
174  foreach (const QVariant &value, values)
175  query.bindValue(k++, value);
176 
177  return query.exec();
178 }
