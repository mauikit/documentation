246 void FMList::sortList()
247 {
248  FMH::MODEL_KEY key = static_cast<FMH::MODEL_KEY>(this->sort);
249  auto index = 0;
250 
251  if(this->foldersFirst)
252  {
253  qSort(this->list.begin(), this->list.end(), [](const FMH::MODEL& e1, const FMH::MODEL& e2) -> bool
254  {
255  const auto key = FMH::MODEL_KEY::MIME;
256  if(e1[key] == "inode/directory")
257  return true;
258 
259  return false;
260  });
261 
262  for(auto item : this->list)
263  if(item[FMH::MODEL_KEY::MIME] == "inode/directory")
264  index++;
265  else break;
266 
267  qSort(this->list.begin(),this->list.begin() + index, [key](const FMH::MODEL& e1, const FMH::MODEL& e2) -> bool
268  {
269  auto role = key;
270 
271  switch(role)
272  {
273  case FMH::MODEL_KEY::SIZE:
274  {
275  if(e1[role].toDouble() > e2[role].toDouble())
276  return true;
277  break;
278  }
279 
280  case FMH::MODEL_KEY::MODIFIED:
281  case FMH::MODEL_KEY::DATE:
282  {
283  auto currentTime = QDateTime::currentDateTime();
284 
285  auto date1 = QDateTime::fromString(e1[role], Qt::TextDate);
286  auto date2 = QDateTime::fromString(e2[role], Qt::TextDate);
287 
288  if(date1.secsTo(currentTime) < date2.secsTo(currentTime))
289  return true;
290 
291  break;
292  }
293 
294  case FMH::MODEL_KEY::LABEL:
295  {
296  const auto str1 = QString(e1[role]).toLower();
297  const auto str2 = QString(e2[role]).toLower();
298 
299  if(str1 < str2)
300  return true;
301  break;
302  }
303 
304  default:
305  if(e1[role] < e2[role])
306  return true;
307  }
308 
309  return false;
310  });
311  }
312 
313  qSort(this->list.begin() + index, this->list.end(), [key](const FMH::MODEL& e1, const FMH::MODEL& e2) -> bool
314  {
315  auto role = key;
316 
317  switch(role)
318  {
319  case FMH::MODEL_KEY::MIME:

741 void FMList::search(const QString& query, const QString &path, const bool &hidden, const bool &onlyDirs, const QStringList &filters)
742 {
743  FMH::MODEL_LIST content;
744 
745  if (FM::isDir(path))
746  {
747  QDir::Filters dirFilter;
748 
749  dirFilter = (onlyDirs ? QDir::AllDirs | QDir::NoDotDot | QDir::NoDot :
750  QDir::Files | QDir::AllDirs | QDir::NoDotDot | QDir::NoDot);
751 
752  if(hidden)
753  dirFilter = dirFilter | QDir::Hidden | QDir::System;
754 
755  QDirIterator it (path, filters, dirFilter, QDirIterator::NoIteratorFlags);
756  while (it.hasNext())
757  {
758  auto url = it.next();
759  auto info = it.fileInfo();
