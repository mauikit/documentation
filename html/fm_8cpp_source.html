59 void FM::init()
60 {
61  this->tag = Tagging::getInstance();
62  this->sync = new Syncing(this);
63  connect(this->sync, &Syncing::listReady, [this](const FMH::MODEL_LIST &list, const QString &url)
64  {
65  emit this->cloudServerContentReady(list, url);
66  });
67 
68  connect(this->sync, &Syncing::itemReady, [this](const FMH::MODEL &item, const QString &url, const Syncing::SIGNAL_TYPE &signalType)
69  {
70  switch(signalType)
71  {
72  case Syncing::SIGNAL_TYPE::OPEN:
73  this->openUrl(item[FMH::MODEL_KEY::PATH]);
74  break;
75 
76  case Syncing::SIGNAL_TYPE::DOWNLOAD:
77  emit this->cloudItemReady(item, url);
78  break;
79 
80  case Syncing::SIGNAL_TYPE::COPY:
81  {
82  QVariantMap data;
83  for(auto key : item.keys())
84  data.insert(FMH::MODEL_NAME[key], item[key]);
85 
86  this->copy(QVariantList {data}, this->sync->getCopyTo());
87  break;
88  }
89  default: return;
90  }
91  });
92 
93  connect(this->sync, &Syncing::error, [this](const QString &message)
94  {
95  emit this->warningMessage(message);
96  });
97 
98  connect(this->sync, &Syncing::progress, [this](const int &percent)
99  {
100  emit this->loadProgress(percent);
101  });
102 
103  connect(this->sync, &Syncing::dirCreated, [this](const FMH::MODEL &dir, const QString &url)
104  {
105  emit this->newItem(dir, url);
106  });
107 
108  connect(this->sync, &Syncing::uploadReady, [this](const FMH::MODEL &item, const QString &url)
109  {
110  emit this->newItem(item, url);
111  });
112 }

697 bool FM::copyPath(QString sourceDir, QString destinationDir, bool overWriteDirectory)
698 {
699  QDir originDirectory(sourceDir);
700 
701  if (!originDirectory.exists())
702  return false;
703 
704  QDir destinationDirectory(destinationDir);
705 
706  if(destinationDirectory.exists() && !overWriteDirectory)
707  return false;
708  else if(destinationDirectory.exists() && overWriteDirectory)
709  destinationDirectory.removeRecursively();
710 
711  originDirectory.mkpath(destinationDir);
712 
713  foreach(QString directoryName, originDirectory.entryList(QDir::Dirs | QDir::NoDotAndDotDot))
714  {
715  QString destinationPath = destinationDir + "/" + directoryName;
716  originDirectory.mkpath(destinationPath);
717  copyPath(sourceDir + "/" + directoryName, destinationPath, overWriteDirectory);
718  }
719 
720  foreach (QString fileName, originDirectory.entryList(QDir::Files))
721  {
722  QFile::copy(sourceDir + "/" + fileName, destinationDir + "/" + fileName);
723  }
724 
726  QDir finalDestination(destinationDir);
727  finalDestination.refresh();
728 
729  if(finalDestination.exists())
730  return true;
731 
732  return false;
733 }
