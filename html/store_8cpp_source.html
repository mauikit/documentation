108 void Store::contentListResult(Attica::BaseJob* j)
109 {
110  qDebug() << "Content list job returned";
111 
112  FMH::MODEL_LIST list;
113 
114  if (j->metadata().error() == Attica::Metadata::NoError)
115  {
116  Attica::ListJob<Attica::Content> *listJob = static_cast<Attica::ListJob<Attica::Content> *>(j);
117 
118  foreach (const Attica::Content &p, listJob->itemList())
119  {
120  const auto att = p.attributes();
121  list << FMH::MODEL {
122  {FMH::MODEL_KEY::ID, p.id()},
123  {FMH::MODEL_KEY::URL, att[STORE::ATTRIBUTE[STORE::ATTRIBUTE_KEY::DOWNLOAD_LINK]]},
124  {FMH::MODEL_KEY::THUMBNAIL, att[STORE::ATTRIBUTE[STORE::ATTRIBUTE_KEY::PREVIEW_SMALL_1]]},

126  {FMH::MODEL_KEY::THUMBNAIL_2, att[STORE::ATTRIBUTE[STORE::ATTRIBUTE_KEY::PREVIEW_2]]},
127  {FMH::MODEL_KEY::THUMBNAIL_3, att[STORE::ATTRIBUTE[STORE::ATTRIBUTE_KEY::DOWNLOAD_LINK]]},
128  {FMH::MODEL_KEY::LABEL, p.name()},
129  {FMH::MODEL_KEY::OWNER, p.author()},
130  {FMH::MODEL_KEY::LICENSE, p.license()},
131  {FMH::MODEL_KEY::DESCRIPTION, p.description()},
132  {FMH::MODEL_KEY::RATE, QString::number(p.rating())},
133  {FMH::MODEL_KEY::DATE, p.created().toString()},
134  {FMH::MODEL_KEY::MODIFIED, p.updated().toString()},
135  {FMH::MODEL_KEY::TAG, p.tags().join(",")},
136  {FMH::MODEL_KEY::COUNT, QString::number(p.downloads())},
137  {FMH::MODEL_KEY::SOURCE, p.detailpage().toString()}
138  };
139  }
140 
141  emit this->contentReady(list);
142 
143  if (listJob->itemList().isEmpty())
144  {
145  emit this->warning(QLatin1String("No Content found."));
146  }
147 
148  } else if (j->metadata().error() == Attica::Metadata::OcsError)
149  {
150  emit this->warning(QString(QLatin1String("OCS Error: %1")).arg(j->metadata().message()));
151 
152  } else if (j->metadata().error() == Attica::Metadata::NetworkError)
153  {
154  emit this->warning(QString(QLatin1String("Network Error: %1")).arg(j->metadata().message()));
155  } else
156  {
157  emit this->warning(QString(QLatin1String("Unknown Error: %1")).arg(j->metadata().message()));
158  }
